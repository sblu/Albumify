# Albumify Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Train a tiny LoRA-fine-tuned Informative-Drawings model that turns a color album cover into a black-and-white line drawing, deployable on a Raspberry Pi 5 with 1 GB of RAM at ~5 MB on disk.

**Architecture:** Fine-tune the published Informative-Drawings generator (~4.4 M params, 17 MB FP32) with LoRA-Conv adapters on Conv2d layers, train on 500 (cover, Gemini-generated-line-drawing) pairs sourced from MusicBrainz/Cover Art Archive + Gemini 3.1 Flash Image Preview, export merged weights to INT8 ONNX (~5 MB), deploy via onnxruntime CLI on laptop and Pi 5. Training on a GCP T4 spot VM in us-central1.

**Tech Stack:** Python 3.11, PyTorch 2.3, ONNX/onnxruntime, google-genai SDK, MusicBrainz/CAA HTTP APIs, GCP Compute Engine, Cover Art Archive.

**Spec:** `docs/superpowers/specs/2026-05-20-albumify-design.md` — read first for design rationale.

**Reference samples (already in repo root):** `AbbeyRoad.jpg`, `AbbeyRoad-SingleLine.png`, `PinkFloydTheWall.jpeg`, `PinkFloydTheWall-SingleLine.png`. These are out-of-distribution test images and must **not** appear in training data.

```python
"""MusicBrainz Web Service v2 client.

Polite rate limit: 1 req/sec (per their docs). Custom User-Agent required.
"""
from __future__ import annotations

import time
from urllib.parse import quote

import requests


class MBNotFound(Exception):
    pass


class MusicBrainzClient:
    BASE = "https://musicbrainz.org/ws/2"

    def __init__(self, user_agent: str, min_interval: float = 1.0):
        self.user_agent = user_agent
        self.min_interval = min_interval
        self._last_request_t = 0.0

    def _throttle(self) -> None:
        wait = self.min_interval - (time.monotonic() - self._last_request_t)
        if wait > 0:
            time.sleep(wait)
        self._last_request_t = time.monotonic()

    def search_release_group(self, artist: str, title: str) -> str | None:
        """Return MBID of the canonical album release-group, or None."""
        query = f'artist:"{artist}" AND releasegroup:"{title}" AND primarytype:album'
        url = f"{self.BASE}/release-group?query={quote(query)}&fmt=json&limit=5"
        self._throttle()
        resp = requests.get(url, headers={"User-Agent": self.user_agent}, timeout=30)
        resp.raise_for_status()
        groups = resp.json().get("release-groups", [])

        # Prefer entries with no secondary types (skip Remix / Compilation / Live / etc.)
        for g in groups:
            if g.get("primary-type") == "Album" and not g.get("secondary-types"):
                return g["id"]
        return None
```

```python
"""Cover Art Archive HTTP client.

Endpoints used:
    https://coverartarchive.org/release-group/{mbid}/front-1200
    https://coverartarchive.org/release-group/{mbid}/front       (fallback)

Behavior:
- Retry up to max_retries on 5xx with exponential backoff.
- Fall back to /front if /front-1200 keeps failing.
- Reject images smaller than min_dim on either axis (CAA returns native size if smaller than requested).
"""
from __future__ import annotations

import time
from io import BytesIO
from typing import Tuple

import requests
from PIL import Image


class CAATooSmall(Exception):
    pass


class CAAClient:
    BASE = "https://coverartarchive.org/release-group"

    def __init__(
        self,
        user_agent: str,
        min_interval: float = 0.5,
        max_retries: int = 3,
        backoff_base: float = 1.0,
        min_dim: int = 256,
    ):
        self.user_agent = user_agent
        self.min_interval = min_interval
        self.max_retries = max_retries
        self.backoff_base = backoff_base
        self.min_dim = min_dim
        self._last = 0.0

    def _throttle(self) -> None:
        wait = self.min_interval - (time.monotonic() - self._last)
        if wait > 0:
            time.sleep(wait)
        self._last = time.monotonic()

    def _get(self, url: str) -> requests.Response:
        self._throttle()
        return requests.get(url, headers={"User-Agent": self.user_agent},
                            allow_redirects=True, timeout=60)

    def fetch_front(self, mbid: str) -> Tuple[bytes, Tuple[int, int]] | None:
        """Fetch the front cover. Returns (bytes, (w, h)) or None on 404."""
        for url in (f"{self.BASE}/{mbid}/front-1200", f"{self.BASE}/{mbid}/front"):
            for attempt in range(self.max_retries):
                resp = self._get(url)
                if resp.status_code == 404:
                    return None  # no cover for this MBID at all
                if 500 <= resp.status_code < 600:
                    time.sleep(self.backoff_base * (2 ** attempt))
                    continue
                resp.raise_for_status()
                img = Image.open(BytesIO(resp.content))
                w, h = img.size
                if min(w, h) < self.min_dim:
                    raise CAATooSmall(f"{mbid}: native {w}x{h} below {self.min_dim}")
                return resp.content, (w, h)
            # exhausted retries at this URL; fall through to next URL
        return None
```

`albumify/fetch_covers.py`:
```python
"""Orchestrator: read albums.json -> resolve MBIDs -> download covers.

Idempotent: rerunning skips already-downloaded files. Writes a report with
per-slug status so misses can be triaged.
"""
from __future__ import annotations

import argparse
import json
from pathlib import Path

from albumify.musicbrainz import MusicBrainzClient
from albumify.caa import CAAClient, CAATooSmall


def run(albums_json: Path, covers_dir: Path, report_path: Path, user_agent: str) -> int:
    """Run fetch. Returns count of NEW covers downloaded this run."""
    albums = json.loads(Path(albums_json).read_text())
    covers_dir = Path(covers_dir)
    covers_dir.mkdir(parents=True, exist_ok=True)
    mb = MusicBrainzClient(user_agent=user_agent)
    caa = CAAClient(user_agent=user_agent)

    new_count = 0
    lines: list[str] = []
    for entry in albums:
        slug = entry["slug"]
        target = covers_dir / f"{slug}.jpg"
        if target.exists():
            lines.append(f"{slug} skip_exists")
            continue
        try:
            mbid = mb.search_release_group(entry["artist"], entry["title"])
        except Exception as exc:
            lines.append(f"{slug} mb_error {exc}")
            continue
        if mbid is None:
            lines.append(f"{slug} no_mbid")
            continue
        try:
            result = caa.fetch_front(mbid)
        except CAATooSmall as exc:
            lines.append(f"{slug} too_small {exc}")
            continue
        except Exception as exc:
            lines.append(f"{slug} caa_error {exc}")
            continue
        if result is None:
            lines.append(f"{slug} no_cover")
            continue
        data, (w, h) = result
        target.write_bytes(data)
        new_count += 1
        lines.append(f"{slug} OK {w}x{h}")

    Path(report_path).write_text("\n".join(lines) + "\n")
    return new_count


def main() -> None:
    p = argparse.ArgumentParser(description="Download album covers from CAA.")
```

```python
"""Generate line-drawing labels for downloaded covers via Gemini.

Default model: gemini-3.1-flash-image-preview at 1024x1024 (1K) output.
Fallback: gemini-2.5-flash-image (Nano Banana, $0.039/image at 1K-equivalent).

Idempotent. Errors are logged, not raised.
"""
from __future__ import annotations

import argparse
import os
import time
from pathlib import Path

from google import genai
from google.genai import types as gtypes


PROMPT = (
    "Create a single line black-and-white line drawing of this image. "
    "White background, black lines only, no shading or fills, "
    "preserving the main subjects and composition of the album cover."
)


def _genai_call(client, model: str, image_bytes: bytes, mime_type: str = "image/jpeg"):
    """Wrapped so tests can patch this single function."""
    image_part = gtypes.Part.from_bytes(data=image_bytes, mime_type=mime_type)
    config = gtypes.GenerateContentConfig(
        response_modalities=["IMAGE"],
        image_config=gtypes.ImageConfig(image_size="1K"),
    )
    return client.models.generate_content(
        model=model,
        contents=[PROMPT, image_part],
        config=config,
    )


def _extract_png(response) -> bytes | None:
    for cand in getattr(response, "candidates", []) or []:
        for part in getattr(cand.content, "parts", []) or []:
            inline = getattr(part, "inline_data", None)
            if inline and inline.data:
                return inline.data
    return None


def run(
    covers_dir: Path,
    labels_dir: Path,
    report_path: Path,
    *,
    model: str,
    api_key: str,
    limit: int | None = None,
    sleep_s: float = 1.0,
) -> int:
    covers_dir = Path(covers_dir)
    labels_dir = Path(labels_dir)
    labels_dir.mkdir(parents=True, exist_ok=True)
    client = genai.Client(api_key=api_key)

    new_count = 0
    lines: list[str] = []
    covers = sorted(covers_dir.glob("*.jpg"))
    if limit is not None:
        covers = covers[:limit]

    for cover in covers:
        slug = cover.stem
        target = labels_dir / f"{slug}.png"
        if target.exists():
            lines.append(f"{slug} skip_exists")
            continue
        try:
            resp = _genai_call(client, model, cover.read_bytes(), "image/jpeg")
        except Exception as exc:
            lines.append(f"{slug} api_error {exc}")
            continue
        png = _extract_png(resp)
        if png is None:
            lines.append(f"{slug} no_image_in_response")
            continue
        target.write_bytes(png)
        new_count += 1
        lines.append(f"{slug} OK {len(png)}B")
        if sleep_s:
            time.sleep(sleep_s)

    Path(report_path).write_text("\n".join(lines) + "\n")
    return new_count


def main() -> None:
    from dotenv import load_dotenv
    load_dotenv()
    p = argparse.ArgumentParser(description="Generate line-drawing labels via Gemini.")
    p.add_argument("--covers", default="data/covers", type=Path)
    p.add_argument("--out", default="data/labels", type=Path)
    p.add_argument("--report", default="data/labels_report.txt", type=Path)
    p.add_argument("--model", default="gemini-3.1-flash-image-preview",
                   choices=["gemini-3.1-flash-image-preview", "gemini-2.5-flash-image"])
    p.add_argument("--limit", type=int, default=None,
                   help="Process at most N covers (for preview checkpoints).")
    p.add_argument("--sleep", type=float, default=1.0)
    args = p.parse_args()
    api_key = os.environ.get("GEMINI_API_KEY")
    if not api_key:
        raise SystemExit("GEMINI_API_KEY missing. Copy .env.example to .env and fill it in.")
    n = run(args.covers, args.out, args.report,
            model=args.model, api_key=api_key, limit=args.limit, sleep_s=args.sleep)
    print(f"Generated {n} new labels. Report: {args.report}")


if __name__ == "__main__":
    main()
```

```python
"""Build train/eval split files (committed manifest).

Stratified by sort order: every Nth slug goes into eval.
"""
from __future__ import annotations

import argparse
from pathlib import Path


def make_splits(covers_dir: Path, labels_dir: Path, out_dir: Path,
                eval_every: int = 10) -> tuple[list[str], list[str]]:
    covers = {p.stem for p in Path(covers_dir).glob("*.jpg")}
    labels = {p.stem for p in Path(labels_dir).glob("*.png")}
    paired = sorted(covers & labels)
    train, evalset = [], []
    for i, slug in enumerate(paired):
        (evalset if (i + 5) % eval_every == 0 else train).append(slug)
    out_dir = Path(out_dir)
    out_dir.mkdir(parents=True, exist_ok=True)
    (out_dir / "train.txt").write_text("\n".join(train) + "\n")
    (out_dir / "eval.txt").write_text("\n".join(evalset) + "\n")
    return train, evalset


def main() -> None:
    p = argparse.ArgumentParser()
    p.add_argument("--covers", default="data/covers", type=Path)
    p.add_argument("--labels", default="data/labels", type=Path)
    p.add_argument("--out", default="data/splits", type=Path)
    p.add_argument("--eval-every", type=int, default=10)
    args = p.parse_args()
    train, evalset = make_splits(args.covers, args.labels, args.out, args.eval_every)
    print(f"Train: {len(train)}  Eval: {len(evalset)}")


if __name__ == "__main__":
    main()
```

```python
"""Paired image-to-image transforms.

Geometric ops are applied identically to (cover, label); color jitter is
input-only. A custom paired implementation avoids the torchvision footgun
of independently-seeded transforms on a pair.
"""
from __future__ import annotations

import random

import torch
import torchvision.transforms.functional as TF
from PIL import Image


def _to_tensor_normalized(img: Image.Image) -> torch.Tensor:
    """[0,255] PIL -> [-1, 1] float32 tensor, CHW."""
    t = TF.to_tensor(img)  # [0,1]
    return t * 2.0 - 1.0


def paired_transform(
    cover: Image.Image,
    label: Image.Image,
    *,
    size: int = 256,
    train: bool = True,
    seed: int | None = None,
    color_jitter_strength: float = 0.2,
) -> tuple[torch.Tensor, torch.Tensor]:
    if seed is not None:
        rng = random.Random(seed)
    else:
        rng = random

    # 1. Resize so that the smaller side is `size * 1.15`, then random crop.
    cover = cover.convert("RGB")
    label = label.convert("L")
    scale_target = int(size * 1.15)
    cover = TF.resize(cover, scale_target, antialias=True)
    label = TF.resize(label, scale_target, antialias=True)

    if train:
        # Random crop to size×size
        w, h = cover.size
        left = rng.randint(0, w - size)
        top = rng.randint(0, h - size)
        cover = TF.crop(cover, top, left, size, size)
        label = TF.crop(label, top, left, size, size)
        # Horizontal flip
        if rng.random() < 0.5:
            cover = TF.hflip(cover); label = TF.hflip(label)
        # Small rotation
        angle = rng.uniform(-5.0, 5.0)
        cover = TF.rotate(cover, angle, fill=[255, 255, 255])
        label = TF.rotate(label, angle, fill=[255])
        # Color jitter on cover ONLY
        if color_jitter_strength > 0:
            s = color_jitter_strength
            cover = TF.adjust_brightness(cover, 1.0 + rng.uniform(-s, s))
            cover = TF.adjust_saturation(cover, 1.0 + rng.uniform(-s, s))
            cover = TF.adjust_contrast(cover, 1.0 + rng.uniform(-s, s))
    else:
        cover = TF.center_crop(cover, [size, size])
        label = TF.center_crop(label, [size, size])

    return _to_tensor_normalized(cover), _to_tensor_normalized(label)


def eval_transform(cover: Image.Image, label: Image.Image, size: int = 256):
    return paired_transform(cover, label, size=size, train=False)
```

```python
"""Informative-Drawings generator architecture.

Reproduction of the Generator from Chan et al., SIGGRAPH 2022 (MIT-licensed
public implementation at github.com/carolineec/informative-drawings).

Encoder -> 9 residual blocks -> Decoder. Used here as the base model for LoRA fine-tuning.
"""
from __future__ import annotations

import torch
import torch.nn as nn


def count_parameters(m: nn.Module) -> int:
    return sum(p.numel() for p in m.parameters())


class _ResidualBlock(nn.Module):
    def __init__(self, ch: int):
        super().__init__()
        self.block = nn.Sequential(
            nn.ReflectionPad2d(1),
            nn.Conv2d(ch, ch, 3),
            nn.InstanceNorm2d(ch),
            nn.ReLU(inplace=True),
            nn.ReflectionPad2d(1),
            nn.Conv2d(ch, ch, 3),
            nn.InstanceNorm2d(ch),
        )

    def forward(self, x):
        return x + self.block(x)


class InformativeDrawings(nn.Module):
    def __init__(self, input_nc: int = 3, output_nc: int = 1,
                 n_residual_blocks: int = 9, ngf: int = 64):
        super().__init__()
        # Encoder
        layers: list[nn.Module] = [
            nn.ReflectionPad2d(3),
            nn.Conv2d(input_nc, ngf, 7),
            nn.InstanceNorm2d(ngf),
            nn.ReLU(inplace=True),
        ]
        # Two strided convs (downsample x4)
        in_features = ngf
        out_features = ngf * 2
        for _ in range(2):
            layers += [
                nn.Conv2d(in_features, out_features, 3, stride=2, padding=1),
                nn.InstanceNorm2d(out_features),
                nn.ReLU(inplace=True),
            ]
            in_features = out_features
            out_features = in_features * 2
        # Residual blocks
        for _ in range(n_residual_blocks):
            layers.append(_ResidualBlock(in_features))
        # Decoder: two transposed convs (upsample x4)
        out_features = in_features // 2
        for _ in range(2):
            layers += [
                nn.ConvTranspose2d(in_features, out_features, 3,
                                   stride=2, padding=1, output_padding=1),
                nn.InstanceNorm2d(out_features),
                nn.ReLU(inplace=True),
            ]
            in_features = out_features
            out_features = in_features // 2
        # Output
        layers += [
            nn.ReflectionPad2d(3),
            nn.Conv2d(in_features, output_nc, 7),
            nn.Tanh(),
        ]
        self.model = nn.Sequential(*layers)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.model(x)
```

`albumify/lora.py`:
```python
"""LoRA adapters for Conv2d layers.

Per Conv2d with kernel kH x kW:
    A: 1x1 conv (in -> rank), initialized zero
    B: kHxkW conv (rank -> out, same padding), initialized normally

Output: y = base_conv(x) + (alpha / rank) * B(A(x))

A is zero-initialized so the adapter contributes nothing at the start of training
(standard LoRA convention). The base conv is frozen.
"""
from __future__ import annotations

from typing import Iterable

import torch
import torch.nn as nn


class LoRAConv2d(nn.Module):
    def __init__(self, base: nn.Conv2d, rank: int = 8, alpha: int = 16):
        super().__init__()
        assert isinstance(base, nn.Conv2d), "LoRAConv2d wraps nn.Conv2d"
        self.base = base
        for p in self.base.parameters():
            p.requires_grad_(False)
        self.rank = rank
        self.alpha = alpha
        self.scale = alpha / rank

        in_ch = base.in_channels
        out_ch = base.out_channels
        k = base.kernel_size
        stride = base.stride
        padding = base.padding

        # A: 1x1 projection down. Stride 1, no padding.
        self.lora_A = nn.Conv2d(in_ch, rank, kernel_size=1, bias=False)
        # B: kxk with matching stride/padding to absorb full receptive field.
        self.lora_B = nn.Conv2d(rank, out_ch, kernel_size=k, stride=stride,
                                padding=padding, bias=False)

        # Init A to zero so initial output == base output
        nn.init.zeros_(self.lora_A.weight)
        nn.init.kaiming_uniform_(self.lora_B.weight, a=5 ** 0.5)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.base(x) + self.scale * self.lora_B(self.lora_A(x))


def apply_lora(model: nn.Module, rank: int = 8, alpha: int = 16) -> None:
    """Recursively replace every nn.Conv2d in `model` with LoRAConv2d.

    Freezes the wrapped base parameters (LoRAConv2d does this internally) and
    leaves all non-Conv2d parameters frozen too (InstanceNorm, biases on
    transposed convs, etc.).
    """
    # First freeze everything
    for p in model.parameters():
        p.requires_grad_(False)

    def _recurse(parent: nn.Module) -> None:
        for name, child in list(parent.named_children()):
            if isinstance(child, nn.Conv2d):
                setattr(parent, name, LoRAConv2d(child, rank=rank, alpha=alpha))
            else:
                _recurse(child)

    _recurse(model)
    # Re-enable grads for new adapter weights (created by apply_lora -> __init__).
    for n, p in model.named_parameters():
        if "lora_A" in n or "lora_B" in n:
            p.requires_grad_(True)


def lora_parameters(model: nn.Module) -> Iterable[nn.Parameter]:
    return [p for n, p in model.named_parameters()
            if ("lora_A" in n or "lora_B" in n) and p.requires_grad]


def merge_lora(model: nn.Module) -> None:
    """Merge LoRA adapters into the wrapped base conv weights, in place.

    After this call, the model is architecturally equivalent to plain
    Informative-Drawings — every LoRAConv2d is replaced with its base nn.Conv2d
    whose weights now equal W0 + scale * B * A.
    """
    def _recurse(parent: nn.Module) -> None:
        for name, child in list(parent.named_children()):
            if isinstance(child, LoRAConv2d):
                merged = _merge_one(child)
                setattr(parent, name, merged)
            else:
                _recurse(child)
    _recurse(model)


@torch.no_grad()
def _merge_one(layer: LoRAConv2d) -> nn.Conv2d:
    """Fold the LoRA adapter (B * A) into the base conv weight.

    Math:
        y_lora[o, h, w] = sum_{r, kh, kw} B[o, r, kh, kw] * A[r, i, 0, 0] * x[i, h+kh, w+kw]
                       = sum_{i, kh, kw} (sum_r B[o, r, kh, kw] * A[r, i, 0, 0]) * x[i, h+kh, w+kw]
        => delta_W[o, i, kh, kw] = scale * sum_r B[o, r, kh, kw] * A[r, i, 0, 0]

    A is shape (rank, in, 1, 1); B is shape (out, rank, kh, kw); base W is (out, in, kh, kw).
    """
    base = layer.base
    A = layer.lora_A.weight              # (rank, in, 1, 1)
    B = layer.lora_B.weight              # (out, rank, kh, kw)
    A_flat = A.view(A.size(0), A.size(1))                   # (rank, in)
    B_flat = B.view(B.size(0), B.size(1), -1)               # (out, rank, kh*kw)
    delta_flat = torch.einsum("orK,ri->oiK", B_flat, A_flat) # (out, in, kh*kw)
    delta = delta_flat.view(B.size(0), A.size(1), *B.shape[-2:]) * layer.scale

    merged = nn.Conv2d(
        in_channels=base.in_channels,
        out_channels=base.out_channels,
        kernel_size=base.kernel_size,
        stride=base.stride,
        padding=base.padding,
        bias=base.bias is not None,
    )
    merged.weight.data.copy_(base.weight.data + delta)
    if base.bias is not None:
        merged.bias.data.copy_(base.bias.data)
    return merged
```

```python
from pathlib import Path


def load_pretrained(ckpt_path: Path | str, **kwargs) -> "InformativeDrawings":
    """Load Informative-Drawings weights from the upstream checkpoint format."""
    path = Path(ckpt_path)
    if not path.exists():
        raise FileNotFoundError(
            f"{path} not found. Download from "

    # Upstream may save under a top-level key; handle both layouts.
    if "model" in state and isinstance(state["model"], dict):
        state = state["model"]
    model.load_state_dict(state)
    return model
```

```python
"""L1 + VGG perceptual loss for image-to-image fine-tuning."""
from __future__ import annotations

import torch
import torch.nn as nn
import torch.nn.functional as F
from torchvision import models


class PerceptualLoss(nn.Module):
    """L1 distance in VGG16 features at relu_2_2 and relu_3_3.

    Inputs are expected in [-1, 1]. The perceptual model expects ImageNet-
    normalized inputs in [0, 1] -> normalize via mean/std internally.

    Single-channel inputs are broadcast to 3 channels before VGG.
    """
    LAYER_IDX = {"relu_2_2": 8, "relu_3_3": 15}
    IMAGENET_MEAN = torch.tensor([0.485, 0.456, 0.406]).view(1, 3, 1, 1)
    IMAGENET_STD = torch.tensor([0.229, 0.224, 0.225]).view(1, 3, 1, 1)

    def __init__(self):
        super().__init__()
        vgg = models.vgg16(weights=models.VGG16_Weights.IMAGENET1K_V1).features
        self.slice = nn.Sequential(*list(vgg.children())[: max(self.LAYER_IDX.values()) + 1])
        for p in self.slice.parameters():
            p.requires_grad_(False)
        self.eval()
        self.register_buffer("mean", self.IMAGENET_MEAN)
        self.register_buffer("std", self.IMAGENET_STD)

    def _to_vgg(self, x: torch.Tensor) -> torch.Tensor:
        # [-1, 1] -> [0, 1]
        x = (x + 1.0) * 0.5
        if x.size(1) == 1:
            x = x.expand(-1, 3, -1, -1)
        return (x - self.mean) / self.std

    def _features(self, x: torch.Tensor) -> list[torch.Tensor]:
        feats = []
        for i, layer in enumerate(self.slice):
            x = layer(x)
            if i in self.LAYER_IDX.values():
                feats.append(x)
        return feats

    def forward(self, pred: torch.Tensor, target: torch.Tensor) -> torch.Tensor:
        f_p = self._features(self._to_vgg(pred))
        f_t = self._features(self._to_vgg(target))
        loss = sum(F.l1_loss(a, b) for a, b in zip(f_p, f_t))
        return loss


_perceptual_singleton: PerceptualLoss | None = None


def l1_and_perceptual(
    pred: torch.Tensor, target: torch.Tensor, *, perceptual_weight: float = 0.1,
) -> tuple[torch.Tensor, dict[str, torch.Tensor]]:
    global _perceptual_singleton
    if _perceptual_singleton is None:
        _perceptual_singleton = PerceptualLoss().to(pred.device)
    elif _perceptual_singleton.mean.device != pred.device:
        _perceptual_singleton.to(pred.device)
    l1 = F.l1_loss(pred, target)
    perc = _perceptual_singleton(pred, target)
    total = l1 + perceptual_weight * perc
    return total, {"l1": l1.detach(), "perceptual": perc.detach()}
```

```python
"""LoRA fine-tuning training loop.

Outputs per run:
    runs/<name>/
        config.json             # all hyperparams + git SHA
        train_log.jsonl         # per-step loss
        eval_log.jsonl          # per-epoch SSIM/LPIPS/L1
        ckpt_epoch_{N}.pt       # LoRA-only weights (rolling 3 + best)
        best.pt                 # symlink/file pointing to lowest-eval-L1
"""
from __future__ import annotations

import argparse
import json
import os
import subprocess
import time
from pathlib import Path

import torch
import torch.nn.functional as F
from torch.utils.data import DataLoader

from albumify.dataset import AlbumDataset
from albumify.losses import l1_and_perceptual
from albumify.lora import apply_lora, lora_parameters
from albumify.model import InformativeDrawings, load_pretrained


def _build_model_for_train(base_ckpt: Path | None, rank: int, alpha: int) -> torch.nn.Module:
    if base_ckpt is None:
        model = InformativeDrawings(n_residual_blocks=9)
    else:
        model = load_pretrained(base_ckpt)
    apply_lora(model, rank=rank, alpha=alpha)
    return model


def _build_model_for_train_tiny(base_ckpt: Path | None, rank: int, alpha: int) -> torch.nn.Module:
    """Tiny variant for the smoke test (3 residual blocks instead of 9)."""
    model = InformativeDrawings(n_residual_blocks=3)
    apply_lora(model, rank=rank, alpha=alpha)
    return model


def _git_sha() -> str:
    try:
        return subprocess.check_output(["git", "rev-parse", "HEAD"]).decode().strip()
    except Exception:
        return "unknown"


def _save_lora_state(model: torch.nn.Module, path: Path) -> None:
    state = {n: p.detach().cpu() for n, p in model.named_parameters()
             if "lora_A" in n or "lora_B" in n}
    torch.save({"lora": state, "config": {"saved_at": time.time()}}, path)


def train(
    *,
    covers_dir: Path, labels_dir: Path, splits_dir: Path, out_dir: Path,
    base_ckpt: Path | None,
    rank: int = 8, alpha: int = 16,
    epochs: int = 50, batch_size: int = 16,
    lr: float = 1e-4, weight_decay: float = 1e-2,
    image_size: int = 256, perceptual_weight: float = 0.1,
    device: str = "cuda", num_workers: int = 4,
    eval_every: int = 5, seed: int = 42,
) -> list[dict]:
    torch.manual_seed(seed)
    out_dir = Path(out_dir); out_dir.mkdir(parents=True, exist_ok=True)
    config = {k: (str(v) if isinstance(v, Path) else v) for k, v in dict(
        rank=rank, alpha=alpha, epochs=epochs, batch_size=batch_size,
        lr=lr, weight_decay=weight_decay, image_size=image_size,
        perceptual_weight=perceptual_weight, device=device, seed=seed,
        git_sha=_git_sha(), base_ckpt=str(base_ckpt) if base_ckpt else None,
    ).items()}
    (out_dir / "config.json").write_text(json.dumps(config, indent=2))

    train_ds = AlbumDataset(covers_dir, labels_dir, splits_dir / "train.txt",
                            train=True, size=image_size)
    eval_ds = AlbumDataset(covers_dir, labels_dir, splits_dir / "eval.txt",
                           train=False, size=image_size)
    train_dl = DataLoader(train_ds, batch_size=batch_size, shuffle=True,
                          num_workers=num_workers, drop_last=False)
    eval_dl = DataLoader(eval_ds, batch_size=batch_size, shuffle=False,
                         num_workers=num_workers)

    model = _build_model_for_train(base_ckpt=base_ckpt, rank=rank, alpha=alpha).to(device)
    optim = torch.optim.AdamW(lora_parameters(model), lr=lr, weight_decay=weight_decay)
    sched = torch.optim.lr_scheduler.CosineAnnealingLR(optim, T_max=epochs)

    history: list[dict] = []
    best_eval_l1 = float("inf")
    train_log = open(out_dir / "train_log.jsonl", "w")
    eval_log = open(out_dir / "eval_log.jsonl", "w")

    for epoch in range(1, epochs + 1):
        model.train()
        train_losses: list[float] = []
        for step, (cov, lab) in enumerate(train_dl):
            cov = cov.to(device, non_blocking=True)
            lab = lab.to(device, non_blocking=True)
            pred = model(cov)
            total, parts = l1_and_perceptual(pred, lab, perceptual_weight=perceptual_weight)
            optim.zero_grad(set_to_none=True)
            total.backward()
            optim.step()
            train_losses.append(total.item())
            train_log.write(json.dumps({
                "epoch": epoch, "step": step, "loss": total.item(),
                "l1": parts["l1"].item(), "perceptual": parts["perceptual"].item(),
            }) + "\n")
        sched.step()
        epoch_train_loss = sum(train_losses) / max(1, len(train_losses))

        # Eval every N epochs (and on epoch 1 to surface smoke-level breakage early).
        eval_l1 = None
        if epoch % eval_every == 0 or epoch == 1:
            model.eval()
            tot = 0.0; n = 0
            with torch.no_grad():
                for cov, lab in eval_dl:
                    cov = cov.to(device); lab = lab.to(device)
                    pred = model(cov)
                    tot += F.l1_loss(pred, lab, reduction="sum").item()
                    n += lab.numel()
            eval_l1 = tot / max(1, n)
            eval_log.write(json.dumps({"epoch": epoch, "eval_l1": eval_l1}) + "\n")
            if eval_l1 < best_eval_l1:
                best_eval_l1 = eval_l1
                _save_lora_state(model, out_dir / "best.pt")

        # Save rolling checkpoint
        _save_lora_state(model, out_dir / f"ckpt_epoch_{epoch}.pt")
        # Keep only last 3 rolling
        rolls = sorted(out_dir.glob("ckpt_epoch_*.pt"), key=lambda p: int(p.stem.split("_")[-1]))
        for old in rolls[:-3]:
            old.unlink()

        history.append({"epoch": epoch, "train_loss": epoch_train_loss, "eval_l1": eval_l1})

    train_log.close(); eval_log.close()
    return history


def main() -> None:
    p = argparse.ArgumentParser()
    p.add_argument("--data", default="data", type=Path)
    p.add_argument("--base-ckpt", default="checkpoints/contour_style.pth", type=Path)
    p.add_argument("--out", required=True, type=Path)
    p.add_argument("--rank", type=int, default=8)
    p.add_argument("--alpha", type=int, default=16)
    p.add_argument("--epochs", type=int, default=50)
    p.add_argument("--batch", type=int, default=16)
    p.add_argument("--lr", type=float, default=1e-4)
    p.add_argument("--size", type=int, default=256)
    p.add_argument("--perceptual", type=float, default=0.1)
    p.add_argument("--device", default="cuda" if torch.cuda.is_available() else "cpu")
    p.add_argument("--workers", type=int, default=4)
    args = p.parse_args()

    train(
        covers_dir=args.data / "covers", labels_dir=args.data / "labels",
        splits_dir=args.data / "splits", out_dir=args.out, base_ckpt=args.base_ckpt,
        rank=args.rank, alpha=args.alpha, epochs=args.epochs, batch_size=args.batch,
        lr=args.lr, image_size=args.size, perceptual_weight=args.perceptual,
        device=args.device, num_workers=args.workers,
    )


if __name__ == "__main__":
    main()
```

```python
"""Quantitative + visual evaluation on the held-out split.

Metrics:
- L1: mean abs diff on [-1, 1] images, scaled back to [0, 1].
- SSIM: scikit-image, per-image then mean.
- LPIPS: AlexNet backbone, on the same 3-channel-broadcast input.
"""
from __future__ import annotations

import argparse
import json
from pathlib import Path

import torch
import torch.nn.functional as F
from PIL import Image
from torch.utils.data import DataLoader

from albumify.dataset import AlbumDataset
from albumify.lora import apply_lora, merge_lora
from albumify.model import load_pretrained


_lpips_singleton = None


def _lpips_model(device: str):
    global _lpips_singleton
    if _lpips_singleton is None:
        import lpips
        _lpips_singleton = lpips.LPIPS(net="alex", verbose=False)
    _lpips_singleton = _lpips_singleton.to(device)
    return _lpips_singleton


def _ssim_batch(a: torch.Tensor, b: torch.Tensor) -> float:
    from skimage.metrics import structural_similarity as ssim
    # a, b in [-1, 1], shape (N, 1, H, W)
    a01 = ((a + 1) * 0.5).clamp(0, 1).cpu().numpy()
    b01 = ((b + 1) * 0.5).clamp(0, 1).cpu().numpy()
    scores = []
    for i in range(a01.shape[0]):
        scores.append(ssim(a01[i, 0], b01[i, 0], data_range=1.0))
    return float(sum(scores) / max(1, len(scores)))


def compute_metrics(pred: torch.Tensor, target: torch.Tensor) -> dict[str, float]:
    device = pred.device
    l1 = F.l1_loss((pred + 1) * 0.5, (target + 1) * 0.5).item()
    ssim_val = _ssim_batch(pred, target)
    lp = _lpips_model(str(device))
    # LPIPS expects (N, 3, H, W) in [-1, 1]
    p3 = pred.expand(-1, 3, -1, -1) if pred.size(1) == 1 else pred
    t3 = target.expand(-1, 3, -1, -1) if target.size(1) == 1 else target
    lpips_val = lp(p3, t3).mean().item()
    return {"l1": l1, "ssim": ssim_val, "lpips": lpips_val}


def _load_for_eval(base_ckpt: Path, lora_ckpt: Path, rank: int, alpha: int, device: str):
    model = load_pretrained(base_ckpt)
    apply_lora(model, rank=rank, alpha=alpha)
    state = torch.load(lora_ckpt, map_location="cpu", weights_only=True)
    sd = model.state_dict()
    for k, v in state["lora"].items():
        sd[k] = v
    model.load_state_dict(sd)
    model.to(device).eval()
    return model


@torch.no_grad()
def evaluate(*, covers_dir: Path, labels_dir: Path, splits_dir: Path,
             out_dir: Path, base_ckpt: Path, lora_ckpt: Path,
             rank: int, alpha: int, image_size: int = 256,
             device: str = "cuda") -> dict[str, float]:
    out_dir = Path(out_dir); out_dir.mkdir(parents=True, exist_ok=True)
    ds = AlbumDataset(covers_dir, labels_dir, splits_dir / "eval.txt",
                      train=False, size=image_size)
    dl = DataLoader(ds, batch_size=8, shuffle=False)
    model = _load_for_eval(base_ckpt, lora_ckpt, rank, alpha, device)

    # Compute metrics over the whole eval set, batched.
    all_l1, all_ssim, all_lpips = [], [], []
    grid_rows: list[list[Image.Image]] = []
    for cov, lab in dl:
        cov = cov.to(device); lab = lab.to(device)
        pred = model(cov)
        m = compute_metrics(pred, lab)
        all_l1.append(m["l1"]); all_ssim.append(m["ssim"]); all_lpips.append(m["lpips"])
        # Build rows: cover | label | pred
        for i in range(cov.size(0)):
            def to_img(t, mode):
                t = ((t + 1) * 0.5).clamp(0, 1).cpu()
                if mode == "L":
                    arr = (t[0].numpy() * 255).astype("uint8")
                    return Image.fromarray(arr, mode="L").convert("RGB")
                arr = (t.permute(1, 2, 0).numpy() * 255).astype("uint8")
                return Image.fromarray(arr, mode="RGB")
            grid_rows.append([to_img(cov[i], "RGB"), to_img(lab[i], "L"), to_img(pred[i], "L")])
    means = {
        "l1": sum(all_l1) / len(all_l1),
        "ssim": sum(all_ssim) / len(all_ssim),
        "lpips": sum(all_lpips) / len(all_lpips),
    }
    # Build a single grid image.
    if grid_rows:
        w = image_size * 3
        h = image_size * len(grid_rows)
        grid = Image.new("RGB", (w, h), (255, 255, 255))
        for ri, row in enumerate(grid_rows):
            for ci, im in enumerate(row):
                grid.paste(im, (ci * image_size, ri * image_size))
        grid.save(out_dir / "grid.png")
    (out_dir / "metrics.json").write_text(json.dumps(means, indent=2))
    (out_dir / "report.md").write_text(
        f"# Eval report\n\n"
        f"- L1: {means['l1']:.4f}\n- SSIM: {means['ssim']:.4f}\n- LPIPS: {means['lpips']:.4f}\n\n"
        f"![grid](grid.png)\n"
    )
    return means
```

```python
"""Export to ONNX FP32 and INT8 for deployment."""
from __future__ import annotations

import argparse
from pathlib import Path

import numpy as np
import torch
from PIL import Image

from albumify.lora import apply_lora, merge_lora
from albumify.model import load_pretrained


def export_fp32_onnx(model: torch.nn.Module, out_path: Path,
                     sample_input_shape: tuple[int, ...] = (1, 3, 256, 256)) -> None:
    model.eval()
    merge_lora(model)  # in-place: replace LoRA wrappers with merged plain Conv2d
    dummy = torch.randn(*sample_input_shape)
    torch.onnx.export(
        model, dummy, str(out_path),
        opset_version=17,
        input_names=["input"], output_names=["output"],
        dynamic_axes={"input": {0: "batch"}, "output": {0: "batch"}},
        do_constant_folding=True,
    )


def quantize_int8(fp32_path: Path, int8_path: Path, calibration_images: list[Path],
                  image_size: int = 256) -> None:
    """Static INT8 quantization with MinMax calibration."""
    from onnxruntime.quantization import quantize_static, QuantType, CalibrationDataReader

    class Reader(CalibrationDataReader):
        def __init__(self, images: list[Path], size: int):
            self.iter = iter(images)
            self.size = size
            self.input_name = None

        def get_next(self):
            try:
                p = next(self.iter)
            except StopIteration:
                return None
            img = Image.open(p).convert("RGB").resize((self.size, self.size))
            arr = np.asarray(img, dtype=np.float32) / 255.0
            arr = (arr * 2 - 1).transpose(2, 0, 1)[None].astype(np.float32)
            return {self.input_name or "input": arr}

    reader = Reader(calibration_images, image_size)
    reader.input_name = "input"
    quantize_static(
        model_input=str(fp32_path),
        model_output=str(int8_path),
        calibration_data_reader=reader,
        quant_format="QOperator",
        weight_type=QuantType.QInt8,
        activation_type=QuantType.QInt8,
    )


def _verify_int8_close_to_fp32(fp32_path: Path, int8_path: Path,
                               test_images: list[Path], image_size: int = 256,
                               max_diff: float = 0.05) -> float:
    import onnxruntime as ort
    sess_fp = ort.InferenceSession(str(fp32_path), providers=["CPUExecutionProvider"])
    sess_q = ort.InferenceSession(str(int8_path), providers=["CPUExecutionProvider"])
    worst = 0.0
    for p in test_images:
        img = Image.open(p).convert("RGB").resize((image_size, image_size))
        arr = np.asarray(img, dtype=np.float32) / 255.0
        x = (arr * 2 - 1).transpose(2, 0, 1)[None].astype(np.float32)
        y_fp = sess_fp.run(None, {"input": x})[0]
        y_q = sess_q.run(None, {"input": x})[0]
        diff = float(np.abs(y_fp - y_q).max())
        worst = max(worst, diff)
    if worst > max_diff:
        raise RuntimeError(f"INT8 numerical drift {worst} exceeds threshold {max_diff}")
    return worst


def main() -> None:
    p = argparse.ArgumentParser()
    p.add_argument("--base-ckpt", default="checkpoints/contour_style.pth", type=Path)
    p.add_argument("--lora", required=True, type=Path)
    p.add_argument("--rank", type=int, default=8)
    p.add_argument("--alpha", type=int, default=16)
    p.add_argument("--out", default="artifacts", type=Path)
    p.add_argument("--calib-dir", default="data/covers", type=Path,
                   help="Calibration images for INT8.")
    p.add_argument("--calib-n", type=int, default=100)
    p.add_argument("--eval-dir", default="data/covers", type=Path)
    p.add_argument("--eval-n", type=int, default=20)
    p.add_argument("--size", type=int, default=256)
    args = p.parse_args()

    args.out.mkdir(parents=True, exist_ok=True)
    model = load_pretrained(args.base_ckpt)
    apply_lora(model, rank=args.rank, alpha=args.alpha)
    state = torch.load(args.lora, map_location="cpu", weights_only=True)
    sd = model.state_dict()
    for k, v in state["lora"].items():
        sd[k] = v
    model.load_state_dict(sd)

    fp32 = args.out / "albumify_fp32.onnx"
    int8 = args.out / "albumify_int8.onnx"
    export_fp32_onnx(model, fp32, sample_input_shape=(1, 3, args.size, args.size))

    calib = sorted(args.calib_dir.glob("*.jpg"))[: args.calib_n]
    quantize_int8(fp32, int8, calib, image_size=args.size)
    test_imgs = sorted(args.eval_dir.glob("*.jpg"))[: args.eval_n]
    drift = _verify_int8_close_to_fp32(fp32, int8, test_imgs, image_size=args.size)
    print(f"Exported: {fp32}, {int8}. Max INT8 drift on {len(test_imgs)} images: {drift:.4f}")


if __name__ == "__main__":
    main()
```

`albumify/infer.py`:
```python
"""Inference CLI: cover image -> line drawing PNG.

This is the only file the Pi runs. No torch, no torchvision — onnxruntime only.
"""
from __future__ import annotations

import argparse
from pathlib import Path

import numpy as np
import onnxruntime as ort
from PIL import Image


def run_inference(*, model: Path, src: Path, dst: Path, size: int = 256,
                  threads: int | None = None) -> None:
    so = ort.SessionOptions()
    if threads is not None:
        so.intra_op_num_threads = threads
    sess = ort.InferenceSession(str(model), sess_options=so,
                                providers=["CPUExecutionProvider"])
    in_name = sess.get_inputs()[0].name

    img = Image.open(src).convert("RGB").resize((size, size))
    arr = np.asarray(img, dtype=np.float32) / 255.0
    x = (arr * 2 - 1).transpose(2, 0, 1)[None].astype(np.float32)
    y = sess.run(None, {in_name: x})[0]
    y01 = ((y[0, 0] + 1) * 0.5).clip(0, 1)
    out = (y01 * 255).astype(np.uint8)
    Image.fromarray(out, mode="L").save(dst)


def main() -> None:
    p = argparse.ArgumentParser(description="Run Albumify inference on one image.")
    p.add_argument("--model", required=True, type=Path)
    p.add_argument("--in", dest="src", required=True, type=Path)
    p.add_argument("--out", dest="dst", required=True, type=Path)
    p.add_argument("--size", type=int, default=256)
    p.add_argument("--threads", type=int, default=None)
    args = p.parse_args()
    run_inference(model=args.model, src=args.src, dst=args.dst,
                  size=args.size, threads=args.threads)
    print(f"Wrote {args.dst}")


if __name__ == "__main__":
    main()
```
